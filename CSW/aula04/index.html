equacaoSegundoGrau :: Float -> Float -> Float -> String
equacaoSegundoGrau a b c
    | delta < 0  = "Nao existem raizes reais"
    | delta == 0 = "Uma raiz: x = " ++ show x
    | otherwise  = "Duas raizes: x1 = " ++ show x1 ++ " e x2 = " ++ show x2
  where
    delta = b^2 - 4*a*c
    x  = (-b) / (2*a)
    x1 = (-b + sqrt delta) / (2*a)
